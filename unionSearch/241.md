## 问题连接
https://www.papamelon.com/problem/241

建立一个3\*N长度的parent数组，a表示与a同类的生物，a+N表示a的猎物，a+2\*N表示a的天敌。

注意，本题的设计要求必须有ran数组进行优化，即在合并的时候把小的树合并到大的数上，否则就会超时

## 代码


```c++
//
// Created by s'c on 2022/5/1.
//
#include "bits/stdc++.h"
using namespace std;
const int MAXN = 200000;
int par[MAXN];
int ran[MAXN];
int N,K,a,b,c;
void init(int n){
    for(int i = 1;i<=n;++i){
        par[i] = i;
        ran[i] = 0;
    }
}

int find(int x){
    if(x==par[x]) return x;
    else return par[x]=find(par[x]); // 返回时更改节点的父节点为祖宗节点，就是路径压缩操作
}

void unite(int x,int y){
    int fx = find(x);
    int fy = find(y);
    if(fx==fy) return;
    if(ran[fx]<ran[fy]){
        par[fx] = fy;
    }else{
        par[fy] = fx;
        if(ran[fx]==ran[fy]) ran[fx]++;
    }
}

int main(){
    cin>>N>>K;
    int res = 0;
    init(3*N);
    while(K-->0){
        scanf("%d%d%d",&a,&b,&c);
        if(b>N||c>N){
            res++;
            continue;
        }
        if(a==2 && b==c){
            res++;
            continue;
        }
        if(a==1){ // bc同一类
            if(find(b)==find(c+N) || find(b)==find(c+2*N)){
                res++;
                continue;
            }
            unite(b,c);
            unite(b+N,c+N);
            unite(b+2*N,c+2*N);
        }else if(a==2){
            if(find(b)==find(c)||find(b)==find(c+N)){
                res++;
                continue;
            }
            unite(b+N,c);
            unite(b+2*N,c+N);
            unite(b,c+2*N);
        }
    }
    printf("%d\n",res);
}

```
